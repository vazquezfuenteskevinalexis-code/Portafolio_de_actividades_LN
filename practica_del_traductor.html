#Crear un progama, que permita traducir un texto al espa√±ol, a 5 idiomas seleccionar "I" para traducir a ingles
#f frances, a aleman, y italiano, p portugues, y si presiona x que cierre el progama.
from deep_translator import GoogleTranslator

texto=input("Ingresa texto:")

while True:
    print("\ningresa una tecla para traducir:")
    print("\ni ‚Üí Ingles")
    print("\nf ‚Üí Franaces")
    print("\na ‚Üí Aleman")
    print("\ny ‚Üí Italiano")
    print("\np ‚Üí Portugues")
    print("\nx ‚Üí Salir")

    tecla=input("Tu opcion es:").lower()
    if tecla == "i":
        traduccion=GoogleTranslator(source="auto", target="en").translate(texto)
        print("La traduccion del ingles es:",traduccion)

    elif tecla == "f":
        traduccion=GoogleTranslator(source="auto", target="fr").translate(texto)
        print("La traduccion del frances es:",traduccion)

    elif tecla == "a":
        traduccion=GoogleTranslator(source="auto", target="de").translate(texto)
        print("La traduccion del aleman es:",traduccion)

    elif tecla == "y":
        traduccion = GoogleTranslator(source="auto", target="it").translate(texto)
        print("La traduccion del italiano es:", traduccion)

    elif tecla == "p":
        traduccion = GoogleTranslator(source="auto", target="pt").translate(texto)
        print("La traduccion del portugues es:", traduccion)

    elif tecla == "x":
        print("Gracias por usar el programa de traducciones‚ò∫‚òª:")
        break

    else:
        print("Opcion no valida")
        continue
