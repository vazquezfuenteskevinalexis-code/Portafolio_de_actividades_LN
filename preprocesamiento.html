import nltk
import re

# Descarga de modelo de Tokenizaci√≥n
# nltk.download('punkt_tab')

# Importamos las funciones de las librer√≠as
from nltk.tokenize import word_tokenize, sent_tokenize

# Ejemplo de texto a tokenizar
texto = ("En la materia de lenguaje natural PLN y vision artificial VA, se desarolla "
        "un portafolio de practicas como marerial de apoyo para los siguientes"
        " semestres de la especialidad, incluyendo practicas y glosario de funciones"
        "por cada pr√°ctica.")

# Como buena pr√°ctica antes de tokenizar, hay que limpiar el texto

# Convertir en min√∫scula
texto_limpio = texto.lower()

# Quitar saltos de l√≠neas y espacios
texto_limpio = texto_limpio.strip()

# Elimina caracteres que no sean letras
texto_limpio = re.sub(r'[^a-z√°√©√≠√≥√∫√º√±\s]', '', texto_limpio)

# TOKENIZAR

# Token por palabras
palabras = word_tokenize(texto_limpio, language='spanish')
print("Token en palabras", palabras)

# Token por oraciones
oraciones = sent_tokenize(texto)
print("Token en oraciones:", oraciones)

print("Texto original: ", texto)
print(texto_limpio)
