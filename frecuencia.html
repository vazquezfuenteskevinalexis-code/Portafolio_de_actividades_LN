frase="Python es divertido y poderoso"

#1. Limpiar el texto(minisculas y sin puntuacion basica)
frase_limpia=frase.replace(" ","").replace(".","").replace(",","").lower()

#2. Dividir en palabras (Tokenizacion)
tokens=frase_limpia.split()

#3. Contar palabras usando un diccionario
frecuencia={}
for token in tokens:
    if token in frecuencia:
        frecuencia[token]+=1
    else:
        frecuencia[token]=1

#4. Mostrar resultados
print(frecuencia)
